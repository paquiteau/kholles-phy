      \draw[dashed] (-6,3,-4) -- (-6,0,-4)  --(-6,0,0)
      (-6,0,-4) -- ++(6,0,0);
    \end{tikzpicture}
  \end{center}
  Déterminer le coefficient de frottement $f$ en fonction de $\alpha$, $\beta_0$ et $\beta_1$.
